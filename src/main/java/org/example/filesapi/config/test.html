// Visualization Script const template = `
Total items
{{totalItems}}
Unique extensions
{{uniqueExtensions}}
Top directory groups
{{uniqueDirectories}}
Counts by file extension
Derived only from top-level items to avoid recursive or repetitive nested task data.
{{#if extensionBars.length}} {{#each extensionBars}}
{{label}}
{{value}}
{{/each}} {{/if}} {{#unless extensionBars.length}}
No file extensions could be derived from the response.
{{/unless}}
Counts by top-level directory
{{#if directoryRows.length}} {{#each directoryRows}} {{/each}} {{/if}} {{#unless directoryRows.length}} {{/unless}}
DirectoryCount
{{label}}{{value}}
No directory groupings could be derived.
Sample top-level items
{{#if sampleRows.length}} {{#each sampleRows}} {{/each}} {{/if}} {{#unless sampleRows.length}} {{/unless}}
IDFile pathExtensionDirectory
{{id}} {{filePath}} {{extension}} {{directory}}
No sample rows available.
`; function createPayload() { let json; try { json = pm.response.json(); } catch (e) { return { totalItems: 0, uniqueExtensions: 0, uniqueDirectories: 0, extensionBars: [], directoryRows: [], sampleRows: [] }; } const items = Array.isArray(json) ? json : (json && Array.isArray(json.items) ? json.items : []); const safeString = (value) => typeof value === 'string' ? value : ''; const getFilePath = (item) => { if (!item || typeof item !== 'object') return ''; return safeString(item.filePath || item.path || item.file || item.name); }; const normalizePath = (filePath) => safeString(filePath).replace(/\\/g, '/').trim(); const getExtension = (filePath) => { const normalized = normalizePath(filePath); const lastSegment = normalized.split('/').filter(Boolean).pop() || ''; if (!lastSegment || !lastSegment.includes('.')) return '(none)'; const ext = lastSegment.split('.').pop(); return ext ? ext.toLowerCase() : '(none)'; }; const getDirectory = (filePath) => { const normalized = normalizePath(filePath); const parts = normalized.split('/').filter(Boolean); if (parts.length <= 1) return '(root)'; return parts[0] || '(root)'; }; const countBy = (values) => { const map = {}; values.forEach((value) => { const key = value || '(unknown)'; map[key] = (map[key] || 0) + 1; }); return Object.keys(map) .map((key) => ({ label: key, value: map[key] })) .sort((a, b) => b.value - a.value || a.label.localeCompare(b.label)); }; const topLevel = items .filter((item) => item && typeof item === 'object') .map((item) => { const filePath = getFilePath(item); return { id: item.id == null ? 'N/A' : String(item.id), filePath: filePath || 'N/A', extension: getExtension(filePath), directory: getDirectory(filePath) }; }); const extensionRows = countBy(topLevel.map((item) => item.extension)).slice(0, 10); const directoryRows = countBy(topLevel.map((item) => item.directory)).slice(0, 10); const maxExtensionCount = extensionRows.reduce((max, row) => Math.max(max, row.value), 0) || 1; const extensionBars = extensionRows.map((row) => ({ label: row.label, value: row.value, percent: Math.round((row.value / maxExtensionCount) * 100) })); return { totalItems: topLevel.length, uniqueExtensions: countBy(topLevel.map((item) => item.extension)).length, uniqueDirectories: countBy(topLevel.map((item) => item.directory)).length, extensionBars, directoryRows, sampleRows: topLevel.slice(0, 12) }; } pm.visualizer.set(template, createPayload());
